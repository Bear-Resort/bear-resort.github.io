<script>
  (function () {
    const redirectUrl = "/bear-rush-new";
    const storageKey = "404-auto-redirect";
    const toggle = document.getElementById("auto-redirect-toggle");
    const yesBtn = document.getElementById("bear-rush-yes-btn");
    const statusText = document.getElementById("redirect-status");
    const countdownEl = document.getElementById("countdown-seconds");
    const countdownWrap = document.getElementById("redirect-countdown");
    const runFrame = document.getElementById("beary-run-frame");
    const runFrames = [
      "/bear-rush-new/img/beary/run-1.png",
      "/bear-rush-new/img/beary/run-2.png",
      "/bear-rush-new/img/beary/run-3.png",
    ];
    const runSequence = [0, 1, 2, 1];
    let secondsLeft = 5;
    let timerId = null;
    let runSeqIndex = 0;
    let runTimerId = null;

    const saved = localStorage.getItem(storageKey);
    if (saved === "false") {
      toggle.checked = false;
    }

    function startRunAnimation() {
      if (runTimerId !== null) return;
      runTimerId = setInterval(function () {
        runFrame.src = runFrames[runSequence[runSeqIndex]];
        runSeqIndex = (runSeqIndex + 1) % runSequence.length;
      }, 120);
    }

    function stopCountdown() {
      if (timerId !== null) {
        clearInterval(timerId);
        timerId = null;
      }
    }

    function redirectNow() {
      stopCountdown();
      window.location.href = redirectUrl;
    }

    function updateCountdownDisplay() {
      countdownEl.textContent = String(secondsLeft);
      if (secondsLeft <= 0) {
        redirectNow();
      }
    }

    function startCountdown() {
      stopCountdown();
      secondsLeft = 5;
      countdownEl.textContent = "5";
      countdownWrap.style.display = "block";
      timerId = setInterval(function () {
        secondsLeft -= 1;
        updateCountdownDisplay();
      }, 1000);
    }

    function updateStatusText(enabled) {
      statusText.textContent = enabled ? "Yes" : "No";
    }

    function setAutoRedirect(enabled) {
      localStorage.setItem(storageKey, enabled ? "true" : "false");
      updateStatusText(enabled);
      if (enabled) {
        startCountdown();
      } else {
        stopCountdown();
        countdownWrap.style.display = "none";
      }
    }

    toggle.addEventListener("change", function () {
      setAutoRedirect(toggle.checked);
    });

    yesBtn.addEventListener("click", redirectNow);

    startRunAnimation();

    if (toggle.checked) {
      updateStatusText(true);
      startCountdown();
    } else {
      updateStatusText(false);
      countdownWrap.style.display = "none";
    }
  })();
</script>
